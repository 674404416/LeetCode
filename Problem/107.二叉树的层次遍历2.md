例如：
给定二叉树 [3,9,20,null,null,15,7],

    3
   / \
  9  20
    /  \
   15   7
返回其自底向上的层次遍历为：

[
  [15,7],
  [9,20],
  [3]
]

class Solution {
    public List<List<Integer>> levelOrderBottom(TreeNode root) {
    	LinkedList<List<Integer>> result=new LinkedList<>();
        LinkedList<List<Integer>> ans=new LinkedList<>();
    	if(root==null)
    	{
    		return result;
    	}
		LinkedList list=new LinkedList<>();
		list.offer(root.val);
    	result.offer(list);
    	Queue<TreeNode> queue =new LinkedList();
    	queue.offer(root);
    	while(!queue.isEmpty())
    	{
    		list=new LinkedList<>();
    		int size=queue.size();
    		for(int i=0;i<size;i++)
    		{
    			TreeNode temp=queue.poll();
    			if(temp.left!=null)
    			{
    				queue.offer(temp.left);
    				list.offer(temp.left.val);
    			}
    			if(temp.right!=null)
    			{
    				queue.offer(temp.right);
    				list.offer(temp.right.val);
    			}
    		}
            if(list.size()!=0)
                result.offer(list);	
    	}
        int n=result.size();
        for(int i=0;i<n;i++)
    	{
    		ans.offerFirst(result.poll());
    	}
    	return ans;
    }
}
